## SQL MIGRATION SCRIPTS

### Migration 0002: Add Likes (ManyToMany) + Comments

Copy and execute in Supabase SQL Editor:

```sql
BEGIN;
--
-- Add field likes to tweet
--
CREATE TABLE "tweet_tweet_likes" (
    "id" bigint NOT NULL PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY, 
    "tweet_id" bigint NOT NULL, 
    "user_id" integer NOT NULL
);
--
-- Create model Comment
--
CREATE TABLE "tweet_comment" (
    "id" bigint NOT NULL PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY, 
    "text" text NOT NULL, 
    "created_at" timestamp with time zone NOT NULL, 
    "tweet_id" bigint NOT NULL, 
    "user_id" integer NOT NULL
);

ALTER TABLE "tweet_tweet_likes" ADD CONSTRAINT "tweet_tweet_likes_tweet_id_user_id_bc3805d3_uniq" 
    UNIQUE ("tweet_id", "user_id");
ALTER TABLE "tweet_tweet_likes" ADD CONSTRAINT "tweet_tweet_likes_tweet_id_d0e176cc_fk_tweet_tweet_id" 
    FOREIGN KEY ("tweet_id") REFERENCES "tweet_tweet" ("id") DEFERRABLE INITIALLY DEFERRED;
ALTER TABLE "tweet_tweet_likes" ADD CONSTRAINT "tweet_tweet_likes_user_id_8eb90af4_fk_auth_user_id" 
    FOREIGN KEY ("user_id") REFERENCES "auth_user" ("id") DEFERRABLE INITIALLY DEFERRED;

CREATE INDEX "tweet_tweet_likes_tweet_id_d0e176cc" ON "tweet_tweet_likes" ("tweet_id");
CREATE INDEX "tweet_tweet_likes_user_id_8eb90af4" ON "tweet_tweet_likes" ("user_id");

ALTER TABLE "tweet_comment" ADD CONSTRAINT "tweet_comment_tweet_id_b7e029aa_fk_tweet_tweet_id" 
    FOREIGN KEY ("tweet_id") REFERENCES "tweet_tweet" ("id") DEFERRABLE INITIALLY DEFERRED;
ALTER TABLE "tweet_comment" ADD CONSTRAINT "tweet_comment_user_id_8c22fbdd_fk_auth_user_id" 
    FOREIGN KEY ("user_id") REFERENCES "auth_user" ("id") DEFERRABLE INITIALLY DEFERRED;

CREATE INDEX "tweet_comment_tweet_id_b7e029aa" ON "tweet_comment" ("tweet_id");
CREATE INDEX "tweet_comment_user_id_8c22fbdd" ON "tweet_comment" ("user_id");

COMMIT;
```

**What this does**:
- ✅ Creates `tweet_comment` table for nested comments
- ✅ Creates `tweet_tweet_likes` join table for ManyToMany relationship
- ✅ Adds foreign keys + indexes

### Migration 0004: Notification Model + Performance Improvements

```sql
BEGIN;
--
-- Create model Notification
--
CREATE TABLE "tweet_notification" (
    "id" bigint NOT NULL PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY, 
    "notification_type" varchar(10) NOT NULL, 
    "is_read" boolean NOT NULL, 
    "created_at" timestamp with time zone NOT NULL
);

--
-- Add field parent to comment (nested replies)
--
ALTER TABLE "tweet_comment" ADD COLUMN "parent_id" bigint NULL 
    CONSTRAINT "tweet_comment_parent_id_3548bc34_fk_tweet_comment_id" 
    REFERENCES "tweet_comment"("id") DEFERRABLE INITIALLY DEFERRED;

SET CONSTRAINTS "tweet_comment_parent_id_3548bc34_fk_tweet_comment_id" IMMEDIATE;

--
-- Add field view_count to tweet
--
ALTER TABLE "tweet_tweet" ADD COLUMN "view_count" integer DEFAULT 0 NOT NULL;
ALTER TABLE "tweet_tweet" ALTER COLUMN "view_count" DROP DEFAULT;

--
-- Create performance indexes
--
CREATE INDEX "tweet_comme_tweet_i_cddc56_idx" ON "tweet_comment" ("tweet_id", "created_at" DESC);
CREATE INDEX "tweet_tweet_user_id_7d1d49_idx" ON "tweet_tweet" ("user_id", "created_at" DESC);
CREATE INDEX "tweet_tweet_created_faaae2_idx" ON "tweet_tweet" ("created_at" DESC);

--
-- Add field comment to notification
--
ALTER TABLE "tweet_notification" ADD COLUMN "comment_id" bigint NOT NULL 
    CONSTRAINT "tweet_notification_comment_id_c9ec6103_fk_tweet_comment_id" 
    REFERENCES "tweet_comment"("id") DEFERRABLE INITIALLY DEFERRED;

SET CONSTRAINTS "tweet_notification_comment_id_c9ec6103_fk_tweet_comment_id" IMMEDIATE;

--
-- Add field user to notification
--
ALTER TABLE "tweet_notification" ADD COLUMN "user_id" integer NOT NULL 
    CONSTRAINT "tweet_notification_user_id_bd0d95b8_fk_auth_user_id" 
    REFERENCES "auth_user"("id") DEFERRABLE INITIALLY DEFERRED;

SET CONSTRAINTS "tweet_notification_user_id_bd0d95b8_fk_auth_user_id" IMMEDIATE;

--
-- Create notification indexes
--
CREATE INDEX "tweet_notif_user_id_1d1c46_idx" ON "tweet_notification" ("user_id", "created_at" DESC);
CREATE INDEX "tweet_notif_user_id_490890_idx" ON "tweet_notification" ("user_id", "is_read");
CREATE INDEX "tweet_comment_parent_id_3548bc34" ON "tweet_comment" ("parent_id");
CREATE INDEX "tweet_notification_comment_id_c9ec6103" ON "tweet_notification" ("comment_id");
CREATE INDEX "tweet_notification_user_id_bd0d95b8" ON "tweet_notification" ("user_id");

COMMIT;
```

**What this does**:
- ✅ Creates `notification` table for comment/reply alerts
- ✅ Adds `parent_id` to comments for nested replies
- ✅ Adds `view_count` tracking to tweets
- ✅ Adds performance indexes on (user, created_at), (created_at), etc.

## 🚨 If Queries Still FAIL After Migrations

### Check RLS Policies (Secondary Issue)

Run in Supabase SQL Editor:

```sql
-- Check if RLS is enabled and if policies exist
SELECT schemaname, tablename, rowsecurity 
FROM pg_tables 
WHERE schemaname = 'public' 
AND tablename IN ('tweet_tweet', 'tweet_comment', 'tweet_notification', 'auth_user');

-- If rowsecurity = true but no policies exist:
-- Either:
-- A) Disable RLS on these tables (for Django backend access):
ALTER TABLE "tweet_tweet" DISABLE ROW LEVEL SECURITY;
ALTER TABLE "tweet_comment" DISABLE ROW LEVEL SECURITY;
ALTER TABLE "tweet_notification" DISABLE ROW LEVEL SECURITY;

-- OR B) Create permissive policies (if RLS required):
CREATE POLICY "Allow all" ON "tweet_tweet" FOR ALL USING (true);
CREATE POLICY "Allow all" ON "tweet_comment" FOR ALL USING (true);
CREATE POLICY "Allow all" ON "tweet_notification" FOR ALL USING (true);
```
